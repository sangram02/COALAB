Find the largest number in each array of size N.


.global _start
_start;
ldr r0, =count
ldr r1, =list
ldr r2,[r0]
sub r2,r2, #1
ldr r3,[r1], #4
L1: ldr r4,[r1],#4
cmp r3,r4
bgt L2
ldr r3,[r1]
L2: add r1,r1, #4
adds r2,r2, #-1
bne L1
ldr r4, =res
str r3, [r4]
exit: b exit
.data
count: .word 5
list: .word 9,8,3,5,3
res: .word 0

Separate Even numbers and odds numbers in an array of size N.


.global_start
_start;
ldr r0, =list
ldr r1, =evennum
ldr r2, =oddnum
ldr r3, =count
ldr r3, [r3]
L1: ldr r4, [r0], #4
and r5,r4, #1
cmp r5, #1
beg odd
str r4, [r2], #4
bal L2
odd: str r4,[r1],#4
L2: subs r3,r3,#1
bne L1
exit: b exit
.data
count: .word 6
list: .word 0x23,0x56,0x76,0x8,0x9,0xfb
oddnum:.word 0,0,0,0,0,0
evennum: .word 0,0,0,0,0,0

Arrange the no. in Ascending order:-


.DATA
COUNT DB 04H

.CODE
MAIN PROC
MOV AX,DATA
MOV DS,AX
MOV CL,COUNT
DEC CL
UP2:MOV CH,CL
LEA SI,VALUE

JNZ UP2
END MAIN
